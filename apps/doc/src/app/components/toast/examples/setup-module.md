import {NgModule} from '@angular/core';
import {ZuiToastModule, zuiToastOptionsProvider} from "@digital-plant/zui-components";

@NgModule({
  imports: [
    // required - import our module
    ZuiToastModule,
  ],

    zuiToastOptionsProvider({
      timer: 1000
    })
  ]
})
